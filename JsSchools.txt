                    document.getElementById("selected category").innerHTML = "Primary Institution Name";

                    document.getElementById("selected category").innerHTML = ("Secondary Institution Name");
                    }

                    document.getElementById("selected category").innerHTML = ("Tertiary Institution Name");
                    }
                else {
                    document.getElementById("selected category").innerHTML = ("Employer Name");
                    }
            }
